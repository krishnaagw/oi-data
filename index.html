<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Real-Time Options Open Interest Graph</title>

    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

</head>

<body>

    <div style="width: 80%; margin: auto;">

        <canvas id="options-oi-chart"></canvas>

    </div>

    <script>

        const apiKey = 'pk_0d4f444531cf4d7e9c4284a990c12509';

        const symbol = 'AAPL'; // Replace with the stock symbol for which you want Options data

        const apiUrl = `https://cloud.iexapis.com/stable/stock/${symbol}/options?token=${apiKey}`;

        async function fetchOptionsOIData() {

            try {

                const response = await fetch(apiUrl);

                const data = await response.json();

                // Extracting Open Interest (OI) data

                const openInterestData = data.map(option => option.openInterest);

                return openInterestData;

            } catch (error) {

                console.error('Error fetching Options OI data:', error);

            }

        }

        const ctx = document.getElementById('options-oi-chart').getContext('2d');

        const optionsOIChart = new Chart(ctx, {

            type: 'line',

            data: {

                labels: [],

                datasets: [{

                    label: 'Options Open Interest',

                    data: [],

                    borderColor: 'rgba(75, 192, 192, 1)',

                    borderWidth: 2,

                    fill: false

                }]

            },

            options: {

                scales: {

                    x: [{

                        type: 'realtime'

                    }],

                    y: {

                        beginAtZero: false

                    }

                }

            }

        });

        async function updateOptionsOIData() {

            const openInterestData = await fetchOptionsOIData();

            optionsOIChart.data.labels = openInterestData.map((_, index) => index + 1);

            optionsOIChart.data.datasets[0].data = openInterestData;

            optionsOIChart.update();

        }

        // Initial data fetch

        updateOptionsOIData();

        // Update data every 60 seconds

        setInterval(updateOptionsOIData, 60000);

    </script>

</body>

</html>
